Instruct the slave to connect to and replicate from the master
  log file and position (coordinates) must be set properly.

-- on master --

show master status;

-- on slave --

mysqldump --single-transaction --all-databases --master-data=111 \
  --host=127.0.0.1 --port=3307 | mysql --host=127.0.0.1

show slave status\G
stop slave;

change master to 
  master_host='127.0.0.1',
  master_port=3307,
  master_user='repl',
  master_log_file='mysql-bin.000001',
  master_log_pos=111;

start slave;
